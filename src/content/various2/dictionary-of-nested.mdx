
## dictionary of nested arrays

```js
const data = [
  {
    id: 1,
    name: "Category A",
    items: [
      {
        id: 2,
        name: "Subcategory A1",
        items: [
          { id: 3, name: "Item A1-1", value: 10 },
          { id: 4, name: "Item A1-2", value: 15 }
        ]
      },
      {
        id: 5,
        name: "Subcategory A2",
        items: [
          { id: 6, name: "Item A2-1", value: 20 },
          { id: 7, name: "Item A2-2", value: 25 }
        ]
      }
    ]
  },
  {
    id: 8,
    name: "Category B",
    items: [
      {
        id: 9,
        name: "Subcategory B1",
        items: [
          { id: 10, name: "Item B1-1", value: 30 },
          { id: 11, name: "Item B1-2", value: 35 }
        ]
      },
      {
        id: 12,
        name: "Subcategory B2",
        items: [
          { id: 13, name: "Item B2-1", value: 40 },
          { id: 14, name: "Item B2-2", value: 45 }
        ]
      }
    ]
  }
];

function createNestedDictionary(data) {
  const dictionary = {};

  data.forEach(category => {
    dictionary[category.id] = { ...category, subcategories: {} };

    category.items.forEach(subcategory => {
      dictionary[category.id].subcategories[subcategory.id] = { ...subcategory, items: {} };

      subcategory.items.forEach(item => {
        dictionary[category.id].subcategories[subcategory.id].items[item.id] = item;
      });
    });
  });

  return dictionary;
}

const nestedDictionary = createNestedDictionary(data);

// Access examples
console.log(nestedDictionary[1].name); // Category A
console.log(nestedDictionary[1].subcategories[5].name); // Subcategory A2
console.log(nestedDictionary[1].subcategories[5].items[7].name); // Item A2-2
console.log(nestedDictionary[8].subcategories[12].items[14].value); // 45
```

function createNestedDictionary(data) {
  return data.reduce((acc, category) => {
    // Add category to accumulator
    acc[category.id] = { ...category, subcategories: {} };

    // Use reduce to create subcategories
    category.items.reduce((subAcc, subcategory) => {
      subAcc[subcategory.id] = { ...subcategory, items: {} };

      // Use reduce to create items
      subcategory.items.reduce((itemAcc, item) => {
        itemAcc[item.id] = item; // Assign item to the accumulator
        return itemAcc; // Return item accumulator
      }, subAcc[subcategory.id].items); // Initialize with subcategory's items

      return subAcc; // Return subcategory accumulator
    }, acc[category.id].subcategories); // Initialize with category's subcategories

    return acc; // Return the main accumulator
  }, {});
}

const nestedDictionary = createNestedDictionary(data);
