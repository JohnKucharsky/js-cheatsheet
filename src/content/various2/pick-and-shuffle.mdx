## pick and shuffle

Prototype of random pick implemented in this app

```js title="RandomPickPlus"
class RandomPickPlus {
  constructor() {
    this.allIDs = [];
    this.shuffledIds = [];
    this.easyIDs = [];
  }

  shuffle(array) {
    for (let i = array.length - 1; i > 0; i--) {
      let j = Math.floor(Math.random() * (i + 1));
      [array[i], array[j]] = [array[j], array[i]];
    }
  }

  onStart(ids) {
    this.allIDs = ids;
    this.shuffledIds = ids.filter(
      (item) => !this.easyIDs.includes(item),
    );
    this.shuffle(this.shuffledIds);
  }

  startAndReset(ids) {
    this.allIDs = ids;
    this.shuffledIds = ids;
    this.shuffle(this.shuffledIds);
  }

  nextEasy(id) {
    this.shuffledIds = this.shuffledIds.filter(
      (item) => item !== id,
    );
    this.easyIDs.push(id);
  }

  next(id) {
    this.shuffledIds = this.shuffledIds.filter(
      (item) => item !== id,
    );
  }

  printAll() {
    console.log({
      allIDs: this.allIDs,
      shuffledIds: this.shuffledIds,
      easyIDs: this.easyIDs,
    });
  }
}

const randomPickPlus: RandomPickPlus = new RandomPickPlus();
randomPickPlus.onStart([1, 2, 3]);
randomPickPlus.next(1);
randomPickPlus.nextEasy(2);
randomPickPlus.next(3);
randomPickPlus.printAll();
// { allIDs: [ 1, 2, 3 ], shuffledIds: [], easyIDs: [ 2 ] }
randomPickPlus.onStart([1, 2, 3]);
randomPickPlus.printAll();
// { allIDs: [ 1, 2, 3 ], shuffledIds: [ 3, 1 ], easyIDs: [ 2 ] }
randomPickPlus.startAndReset([1, 2, 3]);
randomPickPlus.printAll();
// { allIDs: [ 1, 2, 3 ], shuffledIds: [ 1, 2, 3 ], easyIDs: [ 2 ] }
```
