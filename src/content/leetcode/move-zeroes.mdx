
## move zeroes

Move zeroes to the end

```js title="moveZeroes"
function moveZeroes(nums) {
  let left = 0;

  for (let right = 0; right < nums.length; right++) {
    if (nums[right] !== 0) {
      [nums[right], nums[left]] = [nums[left], nums[right]];
      left++;
    }
  }

  return nums;
}

const arr = [0, 1, 0, 3, 12];
console.log(moveZeroes(arr));
```
