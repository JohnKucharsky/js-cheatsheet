
## house robber

**Example 1:** \
**Input:** nums = [1,2,3,1] \
**Output:** 4 \
**Explanation:** Rob house 1 (money = 1) and then rob house 3 (money = 3).
Total amount you can rob = 1 + 3 = 4

**Example 2:** \
**Input:** nums = [2,7,9,3,1] \
**Output:** 12 \
**Explanation:** Rob house 1 (money = 2), rob house 3 (money = 9) and rob house 5 (money = 1).
Total amount you can rob = 2 + 9 + 1 = 12.

```js
function rob(nums) {
  const n = nums.length;

  if (n === 1) {
    return nums[0];
  }

  const dp = Array(n).fill(0);

  dp[0] = nums[0];
  dp[1] = Math.max(nums[0], nums[1]);

  for (let i = 2; i < n; i++) {
    dp[i] = Math.max(dp[i - 1], nums[i] + dp[i - 2]);
  }

  return dp[n - 1];
}

console.log(rob([2, 7, 9, 3, 1]));
```

```js
function robBruteForce(nums, index = 0) {
  if (index >= nums.length) {
    return 0;
  }

  const skipCurrent = robBruteForce(nums, index + 1);
  const robCurrent = nums[index] + robBruteForce(nums, index + 2);

  return Math.max(skipCurrent, robCurrent);
}

console.log(robBruteForce([2, 7, 9, 3, 1])); // 12
```
