
## valid palindrome

```js title="isPalindrome"
function isPalindrome(str) {
    let j = str.length - 1
    for (let i = 0; i < str.length / 2; i++) {
        if (str[i] !== str[j]) {
            return false;
        }
        j--;
    }
    return true;
}

console.log(isPalindrome("abca")); // false
console.log(isPalindrome("racecar")); // true
console.log(isPalindrome("hello")) // false
```
