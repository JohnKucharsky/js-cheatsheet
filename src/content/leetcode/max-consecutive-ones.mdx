
## mox consecutive ones

```js
function findMaxConsecutiveOnes(nums) {
  let max = 0;
  let count = 0;
  for (let i = 0; i < nums.length; i++) {
    if (nums[i] === 1) {
      count++;
    } else {
      max = Math.max(max, count);
      count = 0;
    }
  }
  max = Math.max(max, count);

  return max;
}

console.log(findMaxConsecutiveOnes([1, 1, 0, 1, 1, 1])); // 3
```
