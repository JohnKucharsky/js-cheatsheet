
## contains duplicate

```js
function containsDuplicate(nums) {
  nums.sort((a, b) => a - b);

  for (let i = 1; i < nums.length; i++) {
    if (nums[i] === nums[i - 1]) {
      return true;
    }
  }

  return false;
}

const arr = [1, 1, 1, 3, 3, 4, 3, 2, 4, 2];
console.log(containsDuplicate(arr)); // true
```
