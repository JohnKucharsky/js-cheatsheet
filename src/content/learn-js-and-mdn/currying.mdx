
## currying

Sum with an arbitrary amount of brackets.

```js
function sum(a) {
  let currentSum = a;

  function f(b) {
    if (b === undefined) return currentSum;
    // Handle cases like sum(1)(2)(undefined)
    currentSum += b;
    return f;
  }

  f.valueOf = () => currentSum;
  // For implicit conversion to a number (e.g., +sum(1)(2))

  return f;
}

console.log(+sum(1)(2)); // 3
console.log(+sum(5)(-1)(2)); // 6
console.log(sum(1)(2)()); // 3
console.log(sum(5)(-1)(2)(undefined)); // 6
```
