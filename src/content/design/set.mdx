
## Set js implementation

```js
class CustomSet {
  constructor(iterable) {
    this.items = new Map();

    if (iterable) {
      for (const value of iterable) {
        this.add(value);
      }
    }
  }

  add(value) {
    if (!this.items.has(value)) {
      this.items.set(value, true);
    }
    return this;
  }

  delete(value) {
    return this.items.delete(value);
  }

  has(value) {
    return this.items.has(value);
  }

  clear() {
    this.items.clear();
  }

  get size() {
    return this.items.size;
  }

  // Iterate over the elements in the set
  forEach(callback) {
    for (const value of this.items.keys()) {
      callback(value, this);
    }
  }

  // Convert set to an array
  toArray() {
    return Array.from(this.items.keys());
  }
}

const mySet = new CustomSet([1, 2, 3, 3, 4]);
console.log(mySet.toArray()); // [1, 2, 3, 4]
console.log(mySet.size); // 4

mySet.add(5).add(6);
console.log(mySet.toArray()); // [1, 2, 3, 4, 5, 6]

mySet.delete(3);
console.log(mySet.has(3)); // false

mySet.forEach((value) => console.log(value));
// 1, 2, 4, 5, 6

mySet.clear();
console.log(mySet.size); // 0
```
