
## pick and shuffle

Prototype of random pick implemented in this app

```js
class RandomPickPlus {
  constructor() {
    this.shuffledIds = [];
  }

  shuffle(array) {
    for (let i = array.length - 1; i > 0; i--) {
      let j = Math.floor(Math.random() * (i + 1));
      [array[i], array[j]] = [array[j], array[i]];
    }
  }

  onStart(ids) {
    this.shuffledIds = ids;
    this.shuffle(this.shuffledIds);
  }

  next(id) {
    this.shuffledIds = this.shuffledIds.filter((item) => item !== id);
  }

  printAll() {
    console.log({
      shuffledIds: this.shuffledIds,
    });
  }
}

const randomPickPlus = new RandomPickPlus();
randomPickPlus.onStart(["sort", "map", "reduce"]);
randomPickPlus.next("sort");
randomPickPlus.next("map");
randomPickPlus.printAll();
// { shuffledIds: [ 'reduce' ] }
```
