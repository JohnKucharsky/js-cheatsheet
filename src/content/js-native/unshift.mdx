
## unshift

```js
Array.prototype.unshift = function (...elements) {
  const newArray = [...elements, ...this];

  for (let i = 0; i < newArray.length; i++) {
    this[i] = newArray[i];
  }

  this.length = newArray.length;

  return this.length;
};

const arr = [3, 4, 5];
arr.unshift(1, 2);
console.log(arr); // [1, 2, 3, 4, 5]
console.log(arr.length); // 5
```
